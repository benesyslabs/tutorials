	\begin{align}
	a & = qk + r \space 0 \le r < a \\
	a & = qk' + r' \space 0 \le r' < a \\
	qk + r & = qk' + r' \\
	q(k - k') + (r - r') & = 0 \\
	r' & = r + p \space p \in Z \\
	0 & \le r' < a \\
	0 & \le r + p < a \\
	-p & \le r < a - p \\
	p & = 0 \\
	r' & = r \\
	q(k - k') + (r - r') & = 0 \\
	q(k - k') + (r - r) & = 0 \\
	q(k - k') + 0 & = 0 \\
	q(k - k') & = 0 \\
	k - k' & = 0 \\
	k' & = k
	\end{align}
