	\sum_{i = 1}^{n} i & = 1 + 2 + 3 + ... + n \\
	S & = 1 + 2 + 3 + ... + n \\
	S & = n + (n - 1) + (n - 2) + ... + 1 \\
	2S & = (1 + n) + (2 + (n - 1)) + (3 + (n - 2)) + ... + (n + 1) \\
	& = (1 + n) + (1 + n) + (1 + n) + ... + (1 + n) \\
	& = n(1 + n) \\
	2S & = n(n + 1) \\
	S & = \frac{n(n + 1)}{2} \\
	\sum_{i = 1}^{n} i & = \frac{n(n + 1)}{2}
	\end{align}
